SELECT NULL SQL_ID, NULL CHILD_NUMBER, NULL NAME, NULL VALUE, 
  NULL IS_DEFAULT FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL SQL_ID, NULL CHILD_NUMBER, NULL NAME, NULL VALUE, 
  NULL IS_DEFAULT FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    DECODE(INSTANCE_NUMBER, -1, USERENV('INSTANCE'), INSTANCE_NUMBER) INSTANCE_NUMBER,
    SQL_ID,
    CHILD_NUMBER,
    SORT_BY
  FROM
  ( SELECT
      -1 INSTANCE_NUMBER,
      '77mtwmnd4c4pk' SQL_ID,
      -1 CHILD_NUMBER,
      'NAME' SORT_BY
    FROM
      DUAL
  )
)
SELECT
  O.SQL_ID,
  O.CHILD_NUMBER,
  O.NAME,
  O.VALUE,
  O.ISDEFAULT IS_DEFAULT
FROM
  BASIS_INFO BI,
  GV$SQL_OPTIMIZER_ENV O
WHERE
  O.INST_ID = BI.INSTANCE_NUMBER AND
  O.SQL_ID = BI.SQL_ID AND
  ( BI.CHILD_NUMBER = -1 OR O.CHILD_NUMBER = BI.CHILD_NUMBER )
ORDER BY
  DECODE(BI.SORT_BY, 'NAME', NAME),
  CHILD_NUMBER, 
  NAME
));
