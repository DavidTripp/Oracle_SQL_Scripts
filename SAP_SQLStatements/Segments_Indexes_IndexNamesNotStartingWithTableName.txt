SELECT NULL OWNER, NULL TABLE_NAME, NULL INDEX_NAME, NULL INDEX_TYPE,
  NULL CREATION_DATE FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL INDEX_NAME, NULL INDEX_TYPE,
  NULL CREATION_DATE FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  I.OWNER,
  I.TABLE_NAME,
  I.INDEX_NAME,
  I.INDEX_TYPE,
  TO_CHAR(O.CREATED, 'dd.mm.yyyy hh24:mi.ss') CREATION_DATE
FROM
  BASIS_INFO BI,
  DBA_INDEXES I,
  DBA_OBJECTS O
WHERE
  I.OWNER LIKE BI.OWNER AND
  I.OWNER = O.OWNER AND
  I.INDEX_NAME = O.OBJECT_NAME AND
  O.SUBOBJECT_NAME IS NULL AND
  O.OBJECT_TYPE = 'INDEX' AND
  I.INDEX_TYPE NOT IN ( 'LOB', 'IOT - TOP' ) AND
  I.INDEX_NAME NOT LIKE I.TABLE_NAME || '%'
));
