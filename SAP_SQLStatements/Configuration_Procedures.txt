SELECT NULL OWNER, NULL OBJECT_NAME, NULL PROCEDURE_NAME FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL OBJECT_NAME, NULL PROCEDURE_NAME FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    '%' OWNER,
    '%' PACKAGE_NAME,
    '%DROP%TASK' PROCEDURE_NAME
  FROM
    DUAL
)
SELECT
  P.OWNER,
  P.OBJECT_NAME,
  P.PROCEDURE_NAME
FROM
  BASIS_INFO BI,
  DBA_PROCEDURES P
WHERE
  P.OWNER LIKE BI.OWNER AND
  P.OBJECT_NAME LIKE BI.PACKAGE_NAME AND
  NVL(P.PROCEDURE_NAME, ' ') LIKE BI.PROCEDURE_NAME
ORDER BY
  P.OWNER,
  P.OBJECT_NAME,
  P.PROCEDURE_NAME
));
