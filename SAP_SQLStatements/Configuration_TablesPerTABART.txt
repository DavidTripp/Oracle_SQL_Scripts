SELECT NULL TABART, NULL TABSPACE, NULL NUM_TABLES FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL TABART, NULL TABSPACE, NULL NUM_TABLES FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
SELECT
  T.TABART,
  T.TABSPACE,
  TO_CHAR(DECODE(D.TABART, NULL, 0, COUNT(*)), 999999999) NUM_TABLES
FROM
  TAORA T,
  DD09L D
WHERE
  T.TABART = D.TABART (+)
GROUP BY
  T.TABART,
  D.TABART,
  T.TABSPACE
ORDER BY
  T.TABART,
  T.TABSPACE
));
