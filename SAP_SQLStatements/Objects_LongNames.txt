SELECT   
  /*+ OPT_PARAM('_PUSH_JOIN_UNION_VIEW', 'FALSE') OPT_PARAM('_COMPLEX_VIEW_MERGING', 'FALSE') */
  NULL OWNER,NULL OBJECT_TYPE, NULL OBJECT_NAME, NULL NAME_LENGTH FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER,NULL OBJECT_TYPE, NULL OBJECT_NAME, NULL NAME_LENGTH FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    '%' OBJECT_TYPE,
    '%' OBJECT_NAME,
    'X' ONLY_BRSPACE_RELATED_TYPES,
    29 MIN_LENGTH
  FROM
    DUAL
)
SELECT
  *
FROM
( SELECT
    O.OWNER,
    O.OBJECT_TYPE,
    O.OBJECT_NAME,
    LENGTH(O.OBJECT_NAME) NAME_LENGTH
  FROM
    BASIS_INFO BI,
    DBA_OBJECTS O
  WHERE
    O.OWNER LIKE BI.OWNER AND
    O.OBJECT_TYPE LIKE BI.OBJECT_TYPE AND
    ( BI.ONLY_BRSPACE_RELATED_TYPES = ' ' OR
      O.OBJECT_TYPE IN ( 'TABLE', 'INDEX', 'TRIGGER' )
    ) AND
    O.OBJECT_NAME LIKE BI.OBJECT_NAME AND
    LENGTH(O.OBJECT_NAME) >= BI.MIN_LENGTH
  UNION ALL
  ( SELECT
      C.OWNER,
      'CONSTRAINT' OBJECT_TYPE,
      C.CONSTRAINT_NAME OBJECT_NAME,
      LENGTH(C.CONSTRAINT_NAME) NAME_LENGTH
    FROM
      BASIS_INFO BI,
      DBA_CONSTRAINTS C
    WHERE
      C.OWNER LIKE BI.OWNER AND
      'CONSTRAINT' LIKE BI.OBJECT_TYPE AND
      C.CONSTRAINT_NAME LIKE BI.OBJECT_NAME AND
      LENGTH(C.CONSTRAINT_NAME) >= BI.MIN_LENGTH
  )
)
ORDER BY
  1, 2, 3
));
