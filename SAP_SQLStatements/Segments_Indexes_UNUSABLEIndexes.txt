SELECT NULL OWNER, NULL INDEX_NAME FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL INDEX_NAME FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  DI.OWNER,
  DI.INDEX_NAME
FROM
  BASIS_INFO BI,
  DBA_INDEXES DI
WHERE
  DI.OWNER LIKE BI.OWNER AND
  DI.STATUS = 'UNUSABLE'
ORDER BY
  DI.OWNER,
  DI.INDEX_NAME
));
