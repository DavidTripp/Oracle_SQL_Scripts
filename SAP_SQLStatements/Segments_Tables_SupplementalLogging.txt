SELECT NULL OWNER, NULL TABLE_NAME, NULL LOG_GROUP_TYPE, NULL LOG_GROUP_NAME FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL LOG_GROUP_TYPE, NULL LOG_GROUP_NAME FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    '%' TABLE_NAME
  FROM
    DUAL
)
SELECT
  LG.OWNER,
  LG.TABLE_NAME,
  LG.LOG_GROUP_TYPE,
  LG.LOG_GROUP_NAME
FROM
  BASIS_INFO BI,
  DBA_LOG_GROUPS LG
WHERE
  LG.OWNER LIKE BI.OWNER AND
  LG.TABLE_NAME LIKE BI.TABLE_NAME
ORDER BY
  LG.OWNER,
  LG.TABLE_NAME
));
