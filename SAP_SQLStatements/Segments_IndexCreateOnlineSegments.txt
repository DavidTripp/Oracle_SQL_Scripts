SELECT NULL OWNER, NULL INDEX_NAME, NULL IOT_TABLE_NAME, NULL IOT_INDEX_NAME, 
  NULL IOT_CREATION_DATE FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL INDEX_NAME, NULL IOT_TABLE_NAME, NULL IOT_INDEX_NAME, 
  NULL IOT_CREATION_DATE FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  I.OWNER,
  O2.OBJECT_NAME INDEX_NAME,
  I.TABLE_NAME IOT_TABLE_NAME,
  I.INDEX_NAME IOT_INDEX_NAME,
  TO_CHAR(O.CREATED, 'dd.mm.yyyy hh24:mi.ss') IOT_CREATION_DATE
FROM
  BASIS_INFO BI,
  DBA_INDEXES I,
  DBA_OBJECTS O,
  DBA_OBJECTS O2
WHERE
  I.OWNER LIKE BI.OWNER AND
  I.OWNER = O.OWNER AND
  I.INDEX_NAME = O.OBJECT_NAME AND
  O.SUBOBJECT_NAME IS NULL AND
  O.OBJECT_TYPE = 'INDEX' AND
  I.INDEX_NAME LIKE 'SYS_IOT_TOP%' AND
  I.TABLE_NAME LIKE 'SYS_JOURNAL%' AND
  O2.OBJECT_ID = SUBSTR(I.TABLE_NAME, 13)
));
