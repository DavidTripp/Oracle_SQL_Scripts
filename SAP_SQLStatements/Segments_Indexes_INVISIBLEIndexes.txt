SELECT NULL OWNER, NULL INDEX_NAME, NULL VISIBILITY FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL INDEX_NAME, NULL VISIBILITY FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  I.OWNER,
  I.INDEX_NAME,
  I.VISIBILITY
FROM
  BASIS_INFO BI,
  DBA_INDEXES I
WHERE
  I.OWNER LIKE BI.OWNER AND
  I.VISIBILITY = 'INVISIBLE'
ORDER BY
  I.OWNER,
  I.INDEX_NAME
));
