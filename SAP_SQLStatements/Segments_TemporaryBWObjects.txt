SELECT
  SUBSTR(OBJECT_NAME, 1, 7) PREFIX,
  OBJECT_TYPE,
  TO_CHAR(COUNT(*), 9999999990) NUM_OBJECTS,
  LPAD(TO_CHAR(MIN(CREATED), 'dd.mm.yyyy'), 18) OLDEST_CREATE_DATE
FROM
  DBA_OBJECTS
WHERE
  SUBSTR(OBJECT_NAME, 1, 6) LIKE '/BI0/0%'
GROUP BY
  SUBSTR(OBJECT_NAME, 1, 7),
  OBJECT_TYPE
ORDER BY
  SUBSTR(OBJECT_NAME, 1, 7);
