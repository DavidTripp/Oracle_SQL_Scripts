SELECT NULL OWNER, NULL TABLE_NAME, NULL LAST_ANALYZED FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL LAST_ANALYZED FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  DT.OWNER,
  DT.TABLE_NAME,
  TO_CHAR(DT.LAST_ANALYZED, 'dd.mm.yyyy hh24:mi:ss') LAST_ANALYZED
FROM
  BASIS_INFO BI,
  DBA_TABLES DT
WHERE
  DT.OWNER LIKE BI.OWNER AND
  DT.GLOBAL_STATS = 'NO' AND
  DT.LAST_ANALYZED IS NOT NULL
));
