SELECT NULL TABLE_NAME, NULL PARTITIONING_TYPE, NULL PARTITION_COUNT
  FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL TABLE_NAME, NULL PARTITIONING_TYPE, NULL PARTITION_COUNT
  FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    '/BI_/F%' TABLE_NAME,
    100 PARTITION_THRESHOLD
  FROM
    DUAL
)
SELECT
  PT.TABLE_NAME,
  PT.PARTITIONING_TYPE, 
  TO_CHAR(PT.PARTITION_COUNT, 99999999999990) PARTITION_COUNT
FROM
  BASIS_INFO BI,
  DBA_PART_TABLES PT
WHERE
  PT.OWNER LIKE BI.OWNER AND
  PT.TABLE_NAME LIKE BI.TABLE_NAME AND
  PT.PARTITION_COUNT >= BI.PARTITION_THRESHOLD
ORDER BY
  PT.PARTITION_COUNT DESC
));
