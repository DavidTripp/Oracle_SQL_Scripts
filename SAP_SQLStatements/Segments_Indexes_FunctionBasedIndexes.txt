SELECT NULL OWNER, NULL INDEX_NAME, NULL COLUMN_EXPRESSION FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL INDEX_NAME, NULL COLUMN_EXPRESSION FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  IE.INDEX_OWNER OWNER,
  IE.INDEX_NAME,
  IE.COLUMN_EXPRESSION
FROM
  BASIS_INFO BI,
  DBA_IND_EXPRESSIONS IE
WHERE
  IE.INDEX_OWNER LIKE BI.OWNER
));
