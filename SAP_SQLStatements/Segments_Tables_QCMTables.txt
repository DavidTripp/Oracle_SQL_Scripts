SELECT NULL OWNER, NULL TABLE_NAME, NULL LAST_DDL_TIME, NULL NUM_ROWS,
  NULL BLOCKS FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL LAST_DDL_TIME, NULL NUM_ROWS,
  NULL BLOCKS FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM 
    DUAL
)
SELECT
  DT.OWNER OWNER,
  DT.TABLE_NAME TABLE_NAME,
  TO_CHAR(DO.LAST_DDL_TIME, 'DD.MM.YYYY HH24:MI:SS') LAST_DDL_TIME,
  TO_CHAR(DT.NUM_ROWS, 9999999990) NUM_ROWS,
  TO_CHAR(DT.BLOCKS, 999999990) BLOCKS
FROM
  BASIS_INFO BI,
  DBA_TABLES DT,
  DBA_OBJECTS DO
WHERE
  DT.OWNER LIKE BI.OWNER AND
  DO.OWNER = DT.OWNER AND
  DO.OBJECT_NAME = DT.TABLE_NAME AND
  DO.OBJECT_TYPE = 'TABLE' AND
  DT.TABLE_NAME LIKE 'QCM%'
ORDER BY
  DO.LAST_DDL_TIME
));
