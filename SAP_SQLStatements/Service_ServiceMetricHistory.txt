SELECT NULL START_TIME, NULL SERVICE_NAME, NULL DBTIME_PER_CALL_MS, NULL CPUTIME_PER_CALL_MS, 
NULL NUM_CALL_PER_SEC FROM DUAL WHERE 1=0
UNION ALL
(
SELECT NULL START_TIME, NULL SERVICE_NAME, NULL DBTIME_PER_CALL_MS, NULL CPUTIME_PER_CALL_MS, 
NULL NUM_CALL_PER_SEC FROM DUAL WHERE 1=0
)
UNION ALL
SELECT * FROM (
SELECT
	TO_CHAR(BEGIN_TIME, 'YYYY-MM-DD HH24:MI:SS') START_TIME,
	SERVICE_NAME,
	TO_CHAR(DBTIMEPERCALL/1000,'99990.99') DBTIME_PER_CALL_MS,
	TO_CHAR(CPUPERCALL/1000,'99990.99') CPUTIME_PER_CALL_MS,
	TO_CHAR(CALLSPERSEC,'999999990') NUM_CALL_PER_SEC
FROM
	V$SERVICEMETRIC_HISTORY
WHERE
	SERVICE_NAME NOT LIKE 'SYS%'
ORDER BY
	START_TIME, SERVICE_NAME
)
