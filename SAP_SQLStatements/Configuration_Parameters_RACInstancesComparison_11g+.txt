SELECT NULL NAME, NULL VALUE_1, NULL VALUE_2 FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL NAME, NULL VALUE_1, NULL VALUE_2 FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    1 INSTANCE_NUMBER_1,
    2 INSTANCE_NUMBER_2,
    80 VALUE_LENGTH
  FROM
    DUAL
)
SELECT
  NAME,
  SUBSTR(VALUE_1, 1, BI.VALUE_LENGTH) VALUE_1,
  SUBSTR(VALUE_2, 1, BI.VALUE_LENGTH) VALUE_2
FROM
( SELECT
    NVL(P1.NAME, P2.NAME) NAME,
    DECODE(P1.NAME, NULL, '<doesn''t exist>', P1.VALUE) VALUE_1,
    DECODE(P2.NAME, NULL, '<doesn''t exist>', P2.VALUE) VALUE_2
  FROM
    GV$PARAMETER P1 FULL OUTER JOIN 
    GV$PARAMETER P2 ON
      P1.NAME = P2.NAME
  WHERE
    P1.INST_ID = ( SELECT INSTANCE_NUMBER_1 FROM BASIS_INFO ) AND
    P2.INST_ID = ( SELECT INSTANCE_NUMBER_2 FROM BASIS_INFO ) AND
    ( P1.NAME IS NULL OR
      P2.NAME IS NULL OR
      P1.VALUE != P2.VALUE
    )
),
  BASIS_INFO BI
ORDER BY
  NAME
));
