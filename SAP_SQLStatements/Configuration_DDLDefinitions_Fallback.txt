SELECT
  TRANSLATE(SUBSTR(TEXT, 1 + (I - 1) * 80, 80), CHR(10) || CHR(9), ' ') LINE
FROM
  ( SELECT 
      DBMS_METADATA.GET_DDL('TABLE','CE1BAT0','SAPSR3') TEXT
    FROM
      DUAL ),
  ( SELECT ROWNUM I FROM DBA_HIST_SQLTEXT WHERE ROWNUM <= 1000 ) 
WHERE
  DBMS_LOB.GETLENGTH("TEXT") > (I - 1) * 80;
