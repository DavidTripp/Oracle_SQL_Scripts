SELECT NULL COLUMN_NAME, NULL DATA_TYPE, NULL LENGTH, NULL PREC,
  NULL N, NULL V, NULL H, NULL NUM_DISTINCT,
  NULL BUCKETS, NULL NUM_NULLS, NULL AVG_LEN, NULL LAST_ANALYZED
  FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL COLUMN_NAME, NULL DATA_TYPE, NULL LENGTH, NULL PREC,
  NULL N, NULL V, NULL H, NULL NUM_DISTINCT,
  NULL BUCKETS, NULL NUM_NULLS, NULL AVG_LEN, NULL LAST_ANALYZED
  FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    'CE4SA01' TABLE_NAME,
    'DISTINCT' ORDER_BY               /* DISTINCT, POSITION or NAME */
  FROM
    DUAL
)
SELECT 
  'TABLE_NAME:' COLUMN_NAME, 
  TABLE_NAME DATA_TYPE, 
  NULL LENGTH, 
  NULL PREC,
  NULL N, 
  NULL V, 
  NULL H, 
  NULL NUM_DISTINCT,
  NULL BUCKETS, 
  NULL NUM_NULLS, 
  NULL AVG_LEN, 
  NULL LAST_ANALYZED
FROM 
  BASIS_INFO
UNION ALL
( SELECT 
    'NUM_ROWS:' COLUMN_NAME, 
    TO_CHAR(NUM_ROWS) DATA_TYPE, 
    NULL LENGTH, 
    NULL PREC,
    NULL N, 
    NULL V, 
    NULL H, 
    NULL NUM_DISTINCT,
    NULL BUCKETS, 
    NULL NUM_NULLS, 
    NULL AVG_LEN, 
    NULL LAST_ANALYZED
  FROM 
    BASIS_INFO BI,
    DBA_TABLES T
  WHERE
    T.OWNER LIKE BI.OWNER AND
    T.TABLE_NAME = BI.TABLE_NAME
)
UNION ALL
( SELECT 
    NULL COLUMN_NAME, 
    NULL DATA_TYPE,
    NULL LENGTH, 
    NULL PREC,
    NULL N, 
    NULL V, 
    NULL H, 
    NULL NUM_DISTINCT,
    NULL BUCKETS, 
    NULL NUM_NULLS, 
    NULL AVG_LEN, 
    NULL LAST_ANALYZED
  FROM 
    DUAL
)
UNION ALL
( SELECT
    *
  FROM
  ( SELECT
      COLUMN_NAME,
      DATA_TYPE,
      TO_CHAR(DATA_LENGTH, 99990) LENGTH,
      TO_CHAR(DATA_PRECISION, 990) PREC,
      DECODE(NULLABLE, 'Y', 'X', ' ') N,
      DECODE(VIRTUAL_COLUMN, 'YES', 'X', ' ') V,
      DECODE(HIDDEN_COLUMN, 'YES', 'X', ' ') H,
      TO_CHAR(NUM_DISTINCT, 99999999990) NUM_DISTINCT,
      TO_CHAR(NUM_BUCKETS, 999990) BUCKETS,
      TO_CHAR(NUM_NULLS, 99999990) NUM_NULLS, 
      TO_CHAR(AVG_COL_LEN, 999990) AVG_LEN,
     TO_CHAR(LAST_ANALYZED, 'dd.mm.yyyy hh24:mi:ss') LAST_ANALYZED
    FROM
      BASIS_INFO BI,
      DBA_TAB_COLS TC
    WHERE
      TC.OWNER LIKE BI.OWNER AND
      TC.TABLE_NAME = BI.TABLE_NAME
    ORDER BY
      DECODE(BI.ORDER_BY, 
        'DISTINCT', TC.NUM_DISTINCT, 
        'POSITION', TC.COLUMN_ID, 1) DESC,
      DECODE(BI.ORDER_BY,
        'NAME', TC.COLUMN_NAME, ' ')
  )
)
));
