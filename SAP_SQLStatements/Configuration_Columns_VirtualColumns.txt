SELECT NULL OWNER, NULL TABLE_NAME, NULL COLUMN_NAME FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL COLUMN_NAME FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT /*+ OPT_PARAM('OPTIMIZER_DYNAMIC_SAMPLING', 6) */
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  TC.OWNER,
  TC.TABLE_NAME,
  TC.COLUMN_NAME
FROM
  BASIS_INFO BI,
  DBA_TAB_COLS TC
WHERE
  TC.OWNER LIKE BI.OWNER AND
  TC.VIRTUAL_COLUMN = 'YES'
));
