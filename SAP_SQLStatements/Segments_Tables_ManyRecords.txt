SELECT NULL OWNER, NULL TABLE_NAME, NULL AVG_ROW_LEN, NULL NUM_ROWS_STATS, 
  NULL NUM_ROWS_CURRENT FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL AVG_ROW_LEN, NULL NUM_ROWS_STATS, 
  NULL NUM_ROWS_CURRENT FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    '%' TABLE_NAME,
    50 NUM_RECORDS,
    -1 MIN_NUM_ROWS
  FROM
    DUAL
)
SELECT
  OWNER,
  TABLE_NAME,
  TO_CHAR(AVG_ROW_LEN, 9999999990) AVG_ROW_LEN,
  TO_CHAR(NUM_ROWS_STATS, 9999999999990) NUM_ROWS_STATS,
  TO_CHAR(NUM_ROWS_CURRENT, 999999999999990) NUM_ROWS_CURRENT
FROM
( SELECT
    OWNER,
    TABLE_NAME,
    AVG_ROW_LEN,
    NUM_ROWS NUM_ROWS_STATS,
    NUM_ROWS + DELTA NUM_ROWS_CURRENT,
    NUM_RECORDS
  FROM
  ( SELECT
      T.OWNER,
      T.TABLE_NAME,
      T.AVG_ROW_LEN,
      NVL(T.NUM_ROWS, 0) NUM_ROWS,
      SUM(NVL(TM.INSERTS, 0)) - SUM(NVL(TM.DELETES, 0)) DELTA,
      BI.NUM_RECORDS,
      BI.MIN_NUM_ROWS
    FROM
      BASIS_INFO BI,
      DBA_TABLES T,
      SYS.DBA_TAB_MODIFICATIONS TM
    WHERE
      T.OWNER LIKE BI.OWNER AND
      T.TABLE_NAME LIKE BI.TABLE_NAME AND
      TM.TABLE_OWNER (+) = T.OWNER AND
      TM.TABLE_NAME (+) = T.TABLE_NAME
    GROUP BY
      T.OWNER,
      T.TABLE_NAME,
      T.AVG_ROW_LEN,
      T.NUM_ROWS,
      BI.NUM_RECORDS,
      BI.MIN_NUM_ROWS
  )
  WHERE
    MIN_NUM_ROWS = -1 OR NUM_ROWS + DELTA >= MIN_NUM_ROWS
  ORDER BY
    ( NUM_ROWS + DELTA ) DESC
)
WHERE
  ROWNUM < NUM_RECORDS
));
