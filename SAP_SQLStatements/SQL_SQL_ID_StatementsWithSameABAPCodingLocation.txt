SELECT NULL MODULE, NULL ACTION, NULL SQL_ID, NULL EXECUTIONS, 
  NULL SQL_TEXT FROM DUAL WHERE 1 = 0 
UNION ALL (
SELECT NULL MODULE, NULL ACTION, NULL SQL_ID, NULL EXECUTIONS, 
  NULL SQL_TEXT FROM DUAL WHERE 1 = 0 
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    '45rct7mbpjk3x' SQL_ID
  FROM
    DUAL
),
CODING_LOCATION AS
( SELECT DISTINCT 
    MODULE,
    ACTION
  FROM
  ( SELECT
      S.MODULE,
      S.ACTION
    FROM 
      BASIS_INFO BI,
      V$SQL S
    WHERE 
      S.SQL_ID = BI.SQL_ID
    UNION ALL
    ( SELECT
        S.MODULE,
        S.ACTION
      FROM
        BASIS_INFO BI,
        DBA_HIST_SQLSTAT S
      WHERE 
        S.SQL_ID = BI.SQL_ID
    )
  )
)
SELECT 
  S.MODULE MODULE,
  S.ACTION ACTION,
  S.SQL_ID SQL_ID,
  SUM(S.EXECUTIONS) EXECUTIONS,
  S.SQL_TEXT SQL_TEXT
FROM
  V$SQL S,
  CODING_LOCATION CL
WHERE
  S.MODULE = CL.MODULE AND
  S.ACTION = CL.ACTION
GROUP BY
  S.MODULE,
  S.ACTION,
  S.SQL_ID,
  S.SQL_TEXT
ORDER BY
  S.SQL_TEXT
));
