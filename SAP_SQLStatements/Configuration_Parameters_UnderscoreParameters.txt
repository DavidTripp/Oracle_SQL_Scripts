SELECT NULL NAME, NULL VALUE, NULL DESCRIPTION FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL NAME, NULL VALUE, NULL DESCRIPTION FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT /* DBACOCKPIT initialization required for successful execution (SAP Note 709627) */
    '\_%gby%' PARAMETER_PATTERN,
    '\' ESCAPE_CHARACTER,
    '%' VALUE
  FROM
    DUAL
)
SELECT
  P.KSPPINM NAME,
  V.KSPPSTVL VALUE,
  P.KSPPDESC DESCRIPTION
FROM
  BASIS_INFO BI,
  X$KSPPI P,
  X$KSPPSV V
WHERE
  P.INDX = V.INDX AND
  UPPER(P.KSPPINM) LIKE UPPER(BI.PARAMETER_PATTERN) ESCAPE BI.ESCAPE_CHARACTER AND
  V.KSPPSTVL LIKE BI.VALUE
ORDER BY
  P.KSPPINM
));
