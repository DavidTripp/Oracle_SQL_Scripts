-- May not be executable from within SAP due to a lack of X$ table access permissions
COLUMN NAME FORMAT A40
COLUMN VALUE FORMAT A20
COLUMN DESCRIPTION FORMAT A85
SET LINESIZE 150
SET PAGESIZE 1000
SELECT NULL NAME, NULL VALUE, NULL DESCRIPTION FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL NAME, NULL VALUE, NULL DESCRIPTION FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    '\_%' PARAMETER_PATTERN,
    '\' ESCAPE_CHARACTER,
    '%' VALUE
  FROM
    DUAL
)
SELECT
  P.KSPPINM NAME,
  V.KSPPSTVL VALUE,
  P.KSPPDESC DESCRIPTION
FROM
  BASIS_INFO BI,
  X$KSPPI P,
  X$KSPPSV V
WHERE
  P.INDX = V.INDX AND
  UPPER(P.KSPPINM) LIKE UPPER(BI.PARAMETER_PATTERN) ESCAPE BI.ESCAPE_CHARACTER AND
  V.KSPPSTVL LIKE BI.VALUE
ORDER BY
  P.KSPPINM
));
