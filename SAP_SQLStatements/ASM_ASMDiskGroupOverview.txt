SELECT NULL DG_NUM, NULL DG_NAME, NULL STATE, NULL REDUNDANCY, NULL COMPATIBILITY,
NULL DATABASE_COMPATIBILITY, NULL SP_COMPATIBILITY FROM DUAL WHERE 1=0 
UNION ALL
(SELECT NULL DG_NUM, NULL DG_NAME, NULL STATE, NULL REDUNDANCY, NULL COMPATIBILITY,
NULL DATABASE_COMPATIBILITY, NULL SP_COMPATIBILITY FROM DUAL WHERE 1=0 )
UNION ALL
SELECT * FROM (
WITH BASIS_INFO AS
(
SELECT 
	'X' QUICK /*+ Use V$ASM_DISKGROUP_STAT for quick access, but new disks will not be included */
FROM
	DUAL
)
SELECT * FROM
(
SELECT
	VAD.GROUP_NUMBER DG_NUM,
	VAD.NAME DG_NAME,
	VAD.STATE,
	DECODE(VAD.TYPE,'EXTERN',VAD.TYPE||'(X1)',
					'NORMAL',VAD.TYPE||'(X2)',
					'HIGH',VAD.TYPE||'(X3)') REDUNDANCY,
	VAD.COMPATIBILITY,
	VAD.DATABASE_COMPATIBILITY,
	VP.VALUE SP_COMPATIBILITY
FROM
	V$ASM_DISKGROUP VAD,
	V$PARAMETER VP,
	BASIS_INFO BI
WHERE
	VP.NAME = 'compatible' AND
	BI.QUICK IS NULL
UNION ALL
SELECT
	VAD.GROUP_NUMBER,
	VAD.NAME,
	VAD.STATE,
	DECODE(VAD.TYPE,'EXTERN',VAD.TYPE||'(X1)',
					'NORMAL',VAD.TYPE||'(X2)',
					'HIGH',VAD.TYPE||'(X3)') REDUNDANCY,
	VAD.COMPATIBILITY,
	VAD.DATABASE_COMPATIBILITY,
	VP.VALUE SP_COMPATIBILITY
FROM
	V$ASM_DISKGROUP_STAT VAD,
	V$PARAMETER VP,
	BASIS_INFO BI
WHERE
	VP.NAME = 'compatible' AND
	BI.QUICK = 'X'
)
ORDER BY
	DG_NUM
);
