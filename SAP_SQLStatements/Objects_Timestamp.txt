SELECT 
	NULL OWNER, NULL OBJNAME, NULL OBJTYPE, NULL CREATE_TIME, NULL LAST_DDL_TIME, NULL LAST_ANA_TIME FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT 
	NULL OWNER, NULL OBJNAME, NULL OBJTYPE, NULL CREATE_TIME, NULL LAST_DDL_TIME, NULL LAST_ANA_TIME FROM DUAL WHERE 1 = 0
) UNION ALL ( 
SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
	'SDBAH' TABLE_NAME
  FROM
    DUAL
)
SELECT 
	O.OWNER,
	O.OBJECT_NAME OBJNAME,
	O.OBJECT_TYPE OBJTYPE,
	TO_CHAR(O.CREATED, 'DD.MM.YYYY HH24:MI:SS') CREATE_TIME,
	TO_CHAR(O.LAST_DDL_TIME, 'DD.MM.YYYY HH24:MI:SS') LAST_DDL_TIME,
	TO_CHAR(T.LAST_ANALYZED, 'DD.MM.YYYY HH24:MI:SS') LAST_ANA_TIME
FROM 
	DBA_OBJECTS O, DBA_TABLES T, BASIS_INFO BI
WHERE
	O.OBJECT_NAME = T.TABLE_NAME AND
	T.TABLE_NAME = BI.TABLE_NAME AND
	O.OWNER LIKE BI.OWNER
UNION ALL
SELECT 
	O.OWNER,
	O.OBJECT_NAME OBJNAME,
	O.OBJECT_TYPE OBJTYPE,
	TO_CHAR(O.CREATED, 'DD.MM.YYYY HH24:MI:SS') CREATE_TIME,
	TO_CHAR(O.LAST_DDL_TIME, 'DD.MM.YYYY HH24:MI:SS') LAST_DDL_TIME,
	TO_CHAR(I.LAST_ANALYZED, 'DD.MM.YYYY HH24:MI:SS') LAST_ANA_TIME
FROM 
	DBA_OBJECTS O, DBA_INDEXES I, BASIS_INFO BI
WHERE
	O.OBJECT_NAME = I.INDEX_NAME AND
	I.TABLE_NAME = BI.TABLE_NAME AND
	O.OWNER LIKE BI.OWNER
));
