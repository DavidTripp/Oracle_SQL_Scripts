SELECT NULL OWNER, NULL TABLE_NAME, NULL COLUMN_NAME, NULL SEGMENT_NAME,
  NULL PCTVERSION FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL COLUMN_NAME, NULL SEGMENT_NAME,
  NULL PCTVERSION FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    11 PCTVERSION_MIN_THRESHOLD
  FROM
    DUAL
)
SELECT
  L.OWNER,
  L.TABLE_NAME,
  L.COLUMN_NAME,
  L.SEGMENT_NAME,
  L.PCTVERSION
FROM
  BASIS_INFO BI,
  DBA_LOBS L
WHERE
  L.OWNER LIKE BI.OWNER AND
  L.PCTVERSION >= BI.PCTVERSION_MIN_THRESHOLD
ORDER BY
  L.OWNER,
  L.TABLE_NAME
));
