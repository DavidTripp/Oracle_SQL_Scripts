SELECT NULL OWNER, NULL TABLE_NAME, NULL CONSTRAINT_NAME, NULL STATUS, 
  NULL IOT_TYPE FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL CONSTRAINT_NAME, NULL STATUS,
  NULL IOT_TYPE FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    ' ' ONLY_DISABLED_CONSTRAINTS
  FROM
    DUAL
)
SELECT
  C.OWNER,
  C.TABLE_NAME,
  C.CONSTRAINT_NAME,
  C.STATUS,
  T.IOT_TYPE
FROM
  BASIS_INFO BI,
  DBA_CONSTRAINTS C,
  DBA_TABLES T
WHERE
  C.OWNER = T.OWNER AND
  C.TABLE_NAME = T.TABLE_NAME AND
  CONSTRAINT_TYPE = 'P' AND
  C.OWNER LIKE BI.OWNER AND
  ( BI.ONLY_DISABLED_CONSTRAINTS = ' ' OR C.STATUS = 'DISABLED' )
ORDER BY
  C.OWNER,
  C.TABLE_NAME
));
