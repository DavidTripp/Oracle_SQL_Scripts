SELECT NULL OWNER, NULL SEGMENT_NAME, NULL SEGMENT_TYPE, 
  NULL INITIAL_EXTENT FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL SEGMENT_NAME, NULL SEGMENT_TYPE, 
  NULL INITIAL_EXTENT FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    'ALL' SEGMENT_TYPE,     /* ALL, INDEX, TABLE */
    100 INITIAL_THRESHOLD_MB
  FROM
    DUAL
)
SELECT
  OWNER,
  SEGMENT_NAME,
  SEGMENT_TYPE,
  TO_CHAR(INITIAL_EXTENT / 1024 / 1024, 9999999990.99) INITIAL_EXTENT
FROM
( SELECT
    T.OWNER,
    T.TABLE_NAME SEGMENT_NAME,
    'TABLE' SEGMENT_TYPE,
    T.INITIAL_EXTENT
  FROM 
    BASIS_INFO BI,
    DBA_TABLES T
  WHERE
    T.OWNER LIKE BI.OWNER AND
    BI.SEGMENT_TYPE IN ('ALL', 'TABLE') AND
    T.INITIAL_EXTENT / 1024 / 1024 >= BI.INITIAL_THRESHOLD_MB
  UNION ALL
  ( SELECT
      I.OWNER,
      I.INDEX_NAME SEGMENT_NAME,
      'INDEX' SEGMENT_TYPE,
      I.INITIAL_EXTENT
    FROM 
      BASIS_INFO BI,
      DBA_INDEXES I
    WHERE
      I.OWNER LIKE BI.OWNER AND
      BI.SEGMENT_TYPE IN ('ALL', 'INDEX') AND
      I.INDEX_TYPE != 'LOB' AND
      I.INITIAL_EXTENT / 1024 / 1024 >= BI.INITIAL_THRESHOLD_MB
  )
)
ORDER BY
  SEGMENT_TYPE,
  OWNER,
  SEGMENT_NAME
));
