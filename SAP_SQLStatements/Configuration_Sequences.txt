SELECT NULL SEQUENCE_OWNER, NULL SEQUENCE_NAME FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL SEQUENCE_OWNER, NULL SEQUENCE_NAME FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  S.SEQUENCE_OWNER,
  S.SEQUENCE_NAME
FROM
  BASIS_INFO BI,
  DBA_SEQUENCES S
WHERE
  S.SEQUENCE_OWNER LIKE BI.OWNER AND
  S.SEQUENCE_NAME != 'DDLOG_SEQ'
));
