SELECT NULL OWNER, NULL TABLE_NAME, NULL CREATION_DATE FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL CREATION_DATE FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER,
    'RUPD%' TABLE_NAME
  FROM
    DUAL
)
SELECT
  T.OWNER,
  T.TABLE_NAME,
  TO_CHAR(O.CREATED, 'dd.mm.yyyy hh24:mi:ss') CREATION_DATE
FROM
  BASIS_INFO BI,
  DBA_TABLES T,
  DBA_OBJECTS O
WHERE
  T.OWNER LIKE BI.OWNER AND
  T.TABLE_NAME LIKE BI.TABLE_NAME AND
  T.TEMPORARY = 'Y' AND
  T.OWNER = O.OWNER AND
  T.TABLE_NAME = O.OBJECT_NAME AND
  O.OBJECT_TYPE = 'TABLE'
));
