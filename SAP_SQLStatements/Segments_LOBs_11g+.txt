SELECT NULL OWNER, NULL TABLE_NAME, NULL COLUMN_NAME, NULL LOB_SEGMENT_NAME, NULL SIZE_GB, 
  NULL LOB_INDEX_NAME, NULL PCTVERSION, NULL RETENTION, NULL CACHE, NULL LOGGING, 
  NULL COMPRESSION, NULL IN_ROW, NULL SECUREFILE FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL TABLE_NAME, NULL COLUMN_NAME, NULL LOB_SEGMENT_NAME, NULL SIZE_GB, 
  NULL LOB_INDEX_NAME, NULL PCTVERSION, NULL RETENTION, NULL CACHE, NULL LOGGING, 
  NULL COMPRESSION, NULL IN_ROW, NULL SECUREFILE FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    '%' OWNER,
    '/TXINTF/DOC_XMLS' TABLE_NAME,
    '%' SEGMENT_NAME
  FROM
    DUAL
)
SELECT
  L.OWNER,
  L.TABLE_NAME,
  L.COLUMN_NAME,
  L.SEGMENT_NAME LOB_SEGMENT_NAME,
  TO_CHAR(NVL(S.BYTES, 0) / 1024 / 1024 / 1024, 9990.99) SIZE_GB,
  L.INDEX_NAME LOB_INDEX_NAME,
  L.PCTVERSION,
  L.RETENTION,
  L.CACHE,
  L.LOGGING,
  L.COMPRESSION,
  L.IN_ROW,
  L.SECUREFILE	
FROM
  BASIS_INFO BI,
  DBA_LOBS L,
  DBA_SEGMENTS S
WHERE
  L.OWNER LIKE BI.OWNER AND
  L.TABLE_NAME LIKE BI.TABLE_NAME AND
  L.SEGMENT_NAME = S.SEGMENT_NAME (+)
));
